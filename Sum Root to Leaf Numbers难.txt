    public int sumNumbers(TreeNode root, int ans){
        if(root==null) return 0;
        ans=ans*10+root.val;
        if(root.left==null&&root.right==null) return ans;
        return sumNumbers(root.left,ans)+sumNumbers(root.right,ans);
    }
}
